import { Meta, Story, Canvas } from "@storybook/addon-docs";
import clsx from "clsx";

<!-- Import component and component styles -->

import "./bem.scss";

<!-- Import Docs -->

import README from "./README.md";

export const Template = ({ items }) => `<nav class="utrecht-breadcrumb">
  <ol class="utrecht-breadcrumb__list">${items
    .map(
      ({ href, title, current, focus }) => `
    <li class="utrecht-breadcrumb__item">
      <a class="${clsx(
        "utrecht-breadcrumb__link utrecht-link",
        current && "utrecht-link--current",
        focus && "utrecht-link--focus"
      )}" href="${href}"${current ? 'aria-current="location"' : ""}>${title}</a>
      </li>`
    )
    .join("\n")}
  </ol>
</nav>`;

<Meta
  title="Components/Breadcrumb navigation"
  argTypes={{
    items: {
      description: "Menu items",
      control: "array",
    },
  }}
  parameters={{
    docs: {
      transformSource: (_src, { args }) => Template(args),
    },
    status: "IN DEVELOPMENT",
    notes: {
      UX: README,
    },
  }}
/>
